SELECT NUMERO_U  , 
	CONCAT(SIT ,'-', SITP."DESC")  Situação, 
	NOME, 
	CPF, 
	RG, 
	FONE1, 
	BAIRRO, 
	CIDADE, 
	FONE2, 
	EMAIL, 
	CEP, 
	LOGRADOURO, 
	NUMERO, 
	CONCAT(TIPO_P,'-', TPP."DESC"), 
	DATA_NASC, 
	DATA_CAD, 
	SEXO, 
	ESTADO 
FROM PESSOAS
INNER JOIN SITP ON SITP.SITPCD = PESSOAS.SIT
INNER JOIN TPP ON TPP.TPPCD = PESSOAS.TIPO_P;
